# 環境變數範本
# 請複製此檔案內容到 .env.local

# Supabase 連接資訊
VITE_SUPABASE_URL=https://your-project-ref.supabase.co
VITE_SUPABASE_PUBLISHABLE_KEY=your_anon_key_here

# 專案公開網址（供 Supabase 郵件驗證與 OAuth Redirect 使用）
VITE_PUBLIC_SITE_URL=https://chaos-registry.vercel.app

# App 深鏈設定（供驗證完成後自動開啟 App）
VITE_APP_DEEP_LINK=chaosregistry://auth/verify

# App 下載頁面（若無法開啟 App 時的備援網址）
VITE_APP_DOWNLOAD_URL=https://chaos-registry.vercel.app/download

# 使用說明：
# 1. 創建 .env.local 檔案
# 2. 複製上述內容並替換為您的實際值
# 3. .env.local 已被 .gitignore 保護，不會被提交
